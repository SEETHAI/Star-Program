# Star-Program
Java program to print Star
package starsimpleprogram
public class StarSimpleProgram{
    public static void main(String[] args) {
        int i,j;
        for(i=1;i<=5;i++)
        {
            for(j=1;j<=5;j++)
            {
                System.out.print("*");
            }
            System.out.println();
            }
    }
    
}
