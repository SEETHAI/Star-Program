# Star-Program
Java program to print Star
package starsimpleprogram
public class StarSimpleProgram{
    public static void main(String[] args) {
        int k,j;
        for(k=1;k<=5;k++)
        {
            for(j=1;j<=5;j++)
            {
                System.out.print("*");
            }
            System.out.println();
            }
    }
    
}
